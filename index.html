def hypot(a,b):
    """_summary_

    Args:
        a (_digit_): one side of right angle triangle (not hypotenuse)
        b (_digit_): different side of right angle triangle (not hypotenuse)

    Returns:
        _float_: hypotenuse of right angle triangle
    """
    return (a**2 + b**2)**0.5
